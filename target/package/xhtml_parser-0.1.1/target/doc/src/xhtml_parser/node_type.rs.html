1/// Represents the different types of nodes that can exist in an XML/XHTML document tree.
2///
3/// This enum defines the fundamental node types used in parsing and representing
4/// XML/XHTML documents:
5///
6/// # Variants
7///
8/// * `Head` - The head node of the document tree, representing the document itself
9/// * `Element` - An XML/XHTML element with a name and optional attributes
10///   - `name`: Location information for the element's tag name in the source
11///   - `attributes`: Range information for the element's attributes
12/// * `Text` - A text node containing character data between elements
13///   - Contains location information for the text content in the source
14///
15/// # Examples
16///
17/// ```rust
18/// use crate::xhtml_parser::node_type::NodeType;
19/// use crate::xhtml_parser::defs::{XmlRange, AttributeRange};
20///
21/// // Head node
22/// let head = NodeType::Head;
23///
24/// // Element node with name and attributes
25/// let element = NodeType::Element {
26///     name: 0..5,
27///     attributes: 6..20,
28/// };
29///
30/// // Text node
31/// let text = NodeType::Text(25..35);
32/// ```
33use crate::defs::{AttributeRange, XmlRange};
34use std::fmt::Debug;
35
36#[derive(Clone, PartialEq, Eq)]
37pub enum NodeType {
38    Head,
39    Element {
40        name: XmlRange,
41        attributes: AttributeRange,
42    },
43    Text(XmlRange),
44}
45
46/// Custom implementation of the `Debug` trait for `NodeType`.
47///
48/// This implementation provides human-readable debug output for the different
49/// variants of `NodeType`:
50///
51/// - `Root`: Displays as "Root"
52/// - `Element`: Displays as "Element(name: the_name_range, attributes: the_attributes_range)"
53/// - `Text`: Displays as "Text(text_content_range)"
54///
55/// This is useful for debugging and logging purposes when working with the
56/// node tree structure.
57impl Debug for NodeType {
58    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
59        match self {
60            NodeType::Head => write!(f, "Head"),
61            NodeType::Element { name, attributes } => {
62                write!(f, "Element(name: {:?}, attributes: {:?})", name, attributes)
63            }
64            NodeType::Text(text) => write!(f, "Text({:?})", text),
65        }
66    }
67}
