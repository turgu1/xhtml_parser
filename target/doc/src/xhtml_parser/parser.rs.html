208macro_rules! skip_after_slice_nobreak {
209    ($contents: expr, $p: ident, $max: expr, $slice: expr) => {{
210        if $p < $contents.len() as XmlIdx {
211            let max_pos: XmlIdx = ($p + $max).min($contents.len() as XmlIdx);
212            if let Some(pos) = kmp_find($slice, &$contents[$p as usize..max_pos as usize]) {
213                $p += pos as XmlIdx + $slice.len() as XmlIdx;
214            } else {
215                $p = $contents.len() as XmlIdx; // Move to the end if no more characters match
216            }
217        }
218    }};
219}

306impl Document {
307    /// Displays XML content around an error position for debugging purposes.
308    ///
309    /// This method extracts a 60-character window (30 characters before and after)
310    /// around the error position to provide context when reporting parsing errors.
311    /// The error position is marked with `[*]` in the output.
312    ///
313    /// # Arguments
314    /// * `pos` - The byte position in the XML where the error occurred
315    ///
316    /// # Returns
317    /// A formatted string showing the XML context around the error position    
318    #[inline]
319    fn show_xml_around_error(&self, pos: XmlIdx) -> String {
320        let start = if pos > 30 { pos - 30 } else { 0 };
321        let end = if (pos + 30) < self.xml.len() as XmlIdx {
322            pos + 30
323        } else {
324            self.xml.len() as XmlIdx
325        };
326        format!(
327            "...{}[*]{}...",
328            std::str::from_utf8(&self.xml[start as usize..pos as usize])
329                .unwrap_or("non valid utf-8"),
330            std::str::from_utf8(&self.xml[pos as usize..end as usize]).unwrap_or("non valid utf-8")
331        )
332    }
333
334    /// Creates a standardized parsing error with context information.
335    ///
336    /// This helper method generates a `ParseXmlError::InvalidXml` with a descriptive
337    /// message that includes the error position and surrounding XML content for
338    /// better debugging.
339    ///
340    /// # Arguments
341    /// * `msg` - The error message describing what went wrong
342    /// * `pos` - The byte position in the XML where the error occurred
343    ///
344    /// # Returns
345    /// A `Result` containing the formatted parsing error
346    #[inline]
347    fn invalid(&self, msg: &str, pos: XmlIdx) -> Result<(), ParseXmlError> {
348        Err(ParseXmlError::InvalidXml(format!(
349            "{}. at position {}: {}",
350            msg,
351            pos,
352            self.show_xml_around_error(pos)
353        )))
354    }

356    /// Validates that a closing tag matches its corresponding opening tag.
357    ///
358    /// This method ensures XML well-formedness by verifying that closing tags
359    /// have the same name as their matching opening tags. It extracts the tag
360    /// names from their respective ranges and compares them.
361    ///
362    /// # Arguments
363    /// * `parent_idx` - The node index of the parent element (opening tag)
364    /// * `range` - The byte range containing the closing tag name
365    ///
366    /// # Returns
367    /// `Ok(())` if the tags match, or a parsing error if they don't match
368    /// or if the parent node is not an element
369    #[inline]
370    fn check_closing_tag(&self, parent_idx: NodeIdx, range: XmlRange) -> Result<(), ParseXmlError> {
371        let parent = self.get_node(parent_idx)?;
372        if let NodeType::Element { name, .. } = parent.get_node_type() {
373            let tag_name = self.get_str_from_range(name);
374            let closing_tag = self.get_str_from_range(&range);
375            if tag_name != closing_tag {
376                return self.invalid(
377                    &format!(
378                        "Closing tag '{}' does not match opening tag '{}'",
379                        closing_tag, tag_name
380                    ),
381                    range.start,
382                );
383            }
384        } else {
385            return self.invalid("Expected an element node for closing tag", range.start);
386        }
387        Ok(())
388    }

390    /// Converts a byte slice containing ASCII decimal digits to a String.
391    ///
392    /// This method filters out any non-digit characters and collects only
393    /// the valid ASCII digits (0-9) into a string representation.
394    ///
395    /// # Arguments
396    /// * `s` - A byte slice potentially containing decimal digits
397    ///
398    /// # Returns
399    /// A String containing only the decimal digits found in the input
400    #[inline]
401    fn decimal(&self, s: &[u8]) -> String {
402        // Convert a byte slice containing ASCII digits to a String
403        // This function assumes that the input is valid ASCII digits (0-9).
404        // Using `collect` to gather the digits into a String
405        s.iter()
406            .filter(|&&c| c.is_ascii_digit())
407            .map(|&c| c as char)
408            .collect::<String>()
409    }

412    ///
413    /// This method filters out any non-hexadecimal characters and collects only

431    /// Translates XML escape sequences to their UTF-8 representations.
432    ///
433    /// This method handles three types of escape sequences:
434    /// - Named entities (e.g., `&amp;`, `&lt;`) using the predefined entities map
435    /// - Decimal character references (e.g., `&#65;`)
436    /// - Hexadecimal character references (e.g., `&#x41;`)
437    ///
438    /// The method processes the escape sequence starting after the '&' character
439    /// and writes the UTF-8 bytes to the specified position in the XML buffer.
440    ///
441    /// # Arguments
442    /// * `from` - Starting position after the '&' character
443    /// * `to` - Position where the translated UTF-8 bytes should be written
444    ///
445    /// # Returns
446    /// `Some((next_from, next_to))` if translation succeeds, where `next_from` is
447    /// the position after the semicolon and `next_to` is the position after the
448    /// written UTF-8 bytes. Returns `None` if the escape sequence is invalid.
449    #[inline]
450    fn translate_sequence(&mut self, from: XmlIdx, to: XmlIdx) -> Option<(XmlIdx, XmlIdx)> {
451        let mut from = from;
452        let number = self.xml[from as usize] == HASH;
453        if number {
454            from += 1;
455        }
456        let hex_number = self.xml[from as usize] == X_CHAR;
457        if hex_number {
458            from += 1;
459        }
460        let start = from;
461        scan_until_char_or_nochange!(self.xml, from, SEMI_COLON);
462        if from == start {
463            return None; // No entity found
464        }
465
466        let from_u8 = &self.xml[start as usize..from as usize];
467        let bytes = if number {
468            let value = if hex_number {
469                u32::from_str_radix(self.hexadecimal(from_u8).as_str(), 16).unwrap_or(0)
470            } else {
471                self.decimal(from_u8).parse::<u32>().unwrap_or(0)
472            };
473            if let Some(val) = char::from_u32(value) {
474                val.to_string().into_bytes()
475            } else {
476                // Invalid character
477                return None;
478            }
479        } else {
480            if let Some(entity) = ENTITIES_MAP.get(from_u8) {
481                entity.to_vec()
482            } else {
483                return None;
484            }
485        };
486        let buf = &mut self.xml[to as usize..];
487        let len = bytes.len().min(buf.len());
488        buf[..len].copy_from_slice(&bytes[..len]);
489
490        Some((from + 1, to + len as XmlIdx)) // pass the semicolon
491    }

493    /// Processes XML content by translating escape sequences in-place.
494    ///
495    /// This method scans through the specified range looking for '&' characters
496    /// that indicate escape sequences. When found, it translates the sequences
497    /// to their UTF-8 representations and compacts the content by moving it
498    /// forward in the buffer, effectively replacing sequences with their values.
499    ///
500    /// # Arguments
501    /// * `range` - The byte range in the XML buffer to process
502    ///
503    /// # Returns
504    /// `Some((start, end))` where `start` is the original start position and
505    /// `end` is the new end position after sequence translation and compaction.
506    /// Returns `None` if processing fails.
507    #[inline]
508    fn parse_escapes(&mut self, range: &XmlRange) -> Option<(XmlIdx, XmlIdx)> {
509        let end = range.end;
510        let mut to = range.start;
511        let mut from = range.start;
512
513        loop {
514            let ampersand_pos = self.xml[from as usize..end as usize]
515                .iter()
516                .position(|&c| c == AMPERSAND)
517                .map_or(end, |pos| from + pos as XmlIdx);
518            if ampersand_pos >= end {
519                if from != to {
520                    // Move the tail content to the `to` position
521                    self.xml
522                        .copy_within(from as usize..end as usize, to as usize);
523                }
524                to += end - from;
525                break; // No more '&' found
526            } else {
527                if ampersand_pos > from {
528                    // Move the content before the '&' to the `to` position
529                    self.xml
530                        .copy_within(from as usize..ampersand_pos as usize, to as usize);
531                    to += ampersand_pos - from;
532                }
533                if let Some((new_from, new_to)) = self.translate_sequence(ampersand_pos + 1, to) {
534                    from = new_from;
535                    to = new_to;
536                } else {
537                    // Invalid escape sequence, just skip the '&'
538                    from += 1;
539                }
540            }
541        }
542
543        Some((range.start, to))
544    }

546    /// Removes the namespace prefix from an XML element or attribute name.
547    ///
548    /// This method searches for a colon (`:`) character within the specified range
549    /// and removes everything before it (including the colon itself), effectively
550    /// stripping the namespace prefix. If no colon is found, the original range
551    /// is returned unchanged.
552    ///
553    /// # Arguments
554    /// * `range` - The byte range containing the name with potential namespace prefix
555    ///
556    /// # Returns
557    /// A new range containing only the local name part (without namespace prefix)
558    ///
559    /// # Examples
560    /// - `prefix:localname` becomes `localname`
561    /// - `localname` remains `localname` (no change if no prefix exists)
562    #[inline]
563    fn remove_namespace_prefix(&mut self, range: XmlRange) -> XmlRange {
564        let colon_pos = self.xml[range.start as usize..range.end as usize]
565            .iter()
566            .position(|&c| c == COLON)
567            .map_or(range.end, |pos| range.start + pos as XmlIdx);
568
569        if colon_pos < range.end {
570            // Remove the namespace prefix
571            self.xml.copy_within(
572                (colon_pos + 1) as usize..range.end as usize,
573                range.start as usize,
574            );
575            range.start..(range.start + (range.end - colon_pos - 1))
576        } else {
577            range.start..range.end
578        }
579    }

581    /// Parses the XML document and builds the document tree structure.
582    ///
583    /// This is the main parsing method that implements a state machine to process
584    /// XML content. It handles:
585    /// - Opening and closing tags
586    /// - Self-closing tags
587    /// - Attributes with quoted values
588    /// - Text content with entity translation
589    /// - Comments and processing instructions (bypass)
590    /// - CDATA sections (bypass)
591    /// - DTD and DTD declarations (bypass)
592    ///
593    /// The parser maintains a current parent node and builds the tree by adding
594    /// child nodes as it encounters different XML constructs. It performs
595    /// validation to ensure well-formed XML structure.
596    ///
597    /// # Returns
598    /// `Ok(())` if parsing succeeds, or a `ParseXmlError` if the XML is malformed
599    /// or contains invalid constructs
600    ///
601    /// # Errors
602    /// Returns errors for various malformed XML conditions such as:
603    /// - Mismatched opening and closing tags
604    /// - Invalid character sequences
605    /// - Malformed attributes
606    /// - Unexpected end of document
607    pub(crate) fn parse(&mut self) -> Result<(), ParseXmlError> {
608        let root_idx: NodeIdx = 0;
609
610        //let contents = self.xml.as_slice();
611
612        let mut parent_idx = root_idx;
613        let mut state = State::Start;
614        let mut i: XmlIdx = 0 as XmlIdx;
615        let size = self.xml.len() as XmlIdx;
616
617        loop {
618            state = match state {
619                State::Start => {
620                    scan_until_char!(self.xml, i, LESS_THAN);
621                    State::ReadTag
622                }
623                State::ReadTag => {
624                    if self.xml[i as usize] != LESS_THAN {
625                        return self.invalid("Expected '<' to start a tag", i as XmlIdx);
626                    }
627                    i += 1; // skip first '<'
628                    if i >= size {
629                        State::End
630                    } else {
631                        match self.xml[i as usize] {
632                            SLASH => {
633                                i += 1;
634                                State::ReadTagClose
635                            }
636                            EXCLAMATION_MARK => {
637                                i += 1;
638                                if i < size {
639                                    if self.xml[i as usize] == b'[' {
640                                        i += 1;
641                                        skip_after_slice_nobreak!(
642                                            self.xml,
643                                            i,
644                                            5000,
645                                            &b">".as_slice()
646                                        );
647                                    } else if self.xml[i as usize] == b'-' {
648                                        i += 1;
649                                        skip_after_slice_nobreak!(
650                                            self.xml,
651                                            i,
652                                            5000,
653                                            &b"-->".as_slice()
654                                        );
655                                    } else {
656                                        break;
657                                    }
658                                }
659                                skip_chartype_nobreak!(self.xml, i, Chartype::Space);
660                                if i >= size {
661                                    State::End
662                                } else {
663                                    State::ReadTag
664                                }
665                            }
666                            QUESTION_MARK => {
667                                i += 1;
668                                skip_after_slice_nobreak!(self.xml, i, 500, &b"?>".as_slice());
669                                skip_chartype_nobreak!(self.xml, i, Chartype::Space);
670                                if i >= size {
671                                    State::End
672                                } else {
673                                    State::ReadTag
674                                }
675                            }
676                            _ => State::ReadTagOpen,
677                        }
678                    }
679                }
680                State::ReadTagOpen => {
681                    let start = i;
682                    if CHARTYPE_TABLE[self.xml[i as usize] as usize] & START_SYMBOL == 0 {
683                        return self.invalid(
684                            "Tag name must start with a letter or underscore",
685                            i as XmlIdx,
686                        );
687                    }
688                    i += 1; // skip first char of tag name
689                    skip_chartype!(self.xml, i, SYMBOL);
690
691                    let name_range = if cfg!(feature = "namespace_removal") {
692                        // Remove namespace prefix from attribute name
693                        self.remove_namespace_prefix(start..i)
694                    } else {
695                        // If namespace removal is not enabled, use the original range
696                        start..i
697                    };
698                    let node_idx = self.add_node(
699                        parent_idx,
700                        NodeType::Element {
701                            name: name_range,
702                            attributes: 0..0, // Placeholder for attributes range
703                        },
704                    )?;
705
706                    parent_idx = node_idx;
707                    State::ReadAttribute
708                }
709                State::ReadTagClose => {
710                    let start = i;
711                    skip_chartype!(self.xml, i, Chartype::Space);
712                    scan_until_char_nobreak!(self.xml, i, GREATER_THAN);
713
714                    let name_range = if cfg!(feature = "namespace_removal") {
715                        // Remove namespace prefix from attribute name
716                        self.remove_namespace_prefix(start..i)
717                    } else {
718                        // If namespace removal is not enabled, use the original range
719                        start..i
720                    };
721
722                    self.check_closing_tag(parent_idx, name_range)?;
723                    parent_idx = if parent_idx == 1 {
724                        0
725                    } else {
726                        self.get_parent_idx(parent_idx)?
727                    };
728                    i += 1;
729                    // Todo: in an EPub context not to get rid of spaces here
730                    skip_chartype_nobreak!(self.xml, i, Chartype::Space);
731                    if i >= size {
732                        State::End
733                    } else {
734                        State::ReadContent
735                    }
736                }
737                State::ReadAttribute => {
738                    skip_chartype!(self.xml, i, Chartype::Space);
739                    match self.xml[i as usize] {
740                        SLASH => {
741                            i += 1;
742                            if i >= size {
743                                break;
744                            }
745                            if self.xml[i as usize] != GREATER_THAN {
746                                return self
747                                    .invalid("Expected '>' after '/' in self-closing tag", i);
748                            }
749                            parent_idx = self.get_parent_idx(parent_idx)?;
750
751                            i += 1;
752                            if i >= size {
753                                break;
754                            }
755                            State::ReadContent
756                        }
757                        GREATER_THAN => {
758                            i += 1;
759                            if i >= size {
760                                break;
761                            }
762                            State::ReadContent
763                        }
764                        _ => {
765                            let start = i;
766                            if CHARTYPE_TABLE[self.xml[i as usize] as usize] & START_SYMBOL == 0 {
767                                return self.invalid(
768                                    "Attribute name must start with a letter or underscore",
769                                    i,
770                                );
771                            }
772                            i += 1;
773                            skip_chartype!(self.xml, i, SYMBOL);
774                            if self.xml[i as usize] != EQUAL {
775                                return self.invalid("Attribute must have an '=' sign", i);
776                            }
777                            let end = i;
778                            i += 1; //skip =
779                            if i >= size {
780                                break;
781                            }
782                            let quote = self.xml[i as usize];
783                            if (quote != b'\'') && (quote != b'"') {
784                                return self
785                                    .invalid("Attribute value must be enclosed in quotes", i);
786                            }
787                            i += 1;
788                            let value_start = i;
789                            scan_until_char!(self.xml, i, quote);
790
791                            let value_range = if cfg!(feature = "parse_escapes") {
792                                // Parse and translate entities in the attribute value
793                                if let Some((_new_start, new_end)) =
794                                    self.parse_escapes(&(value_start..i))
795                                {
796                                    value_start..new_end
797                                } else {
798                                    return self
799                                        .invalid("Invalid entity in attribute value", value_start);
800                                }
801                            } else {
802                                // If escape parsing is not enabled, use the original range
803                                value_start..i
804                            };
805
806                            let name_range = if cfg!(feature = "namespace_removal") {
807                                // Remove namespace prefix from attribute name
808                                self.remove_namespace_prefix(start..end)
809                            } else {
810                                // If namespace removal is not enabled, use the original range
811                                start..end
812                            };
813
814                            self.add_attribute(parent_idx, name_range, value_range)?;
815
816                            i += 1;
817                            State::ReadAttribute
818                        }
819                    }
820                }
821                State::ReadContent => {
822                    // i += 1;
823                    // if i >= size {
824                    //     break;
825                    // }
826                    // ToDo: May need to keep spaces
827                    skip_chartype!(self.xml, i, Chartype::Space);
828
829                    let start = i;
830                    scan_until_char!(self.xml, i, LESS_THAN);
831                    if i > start {
832                        let text_range = if cfg!(feature = "parse_escapes") {
833                            // Parse and translate entities in the content
834                            if let Some((_new_start, new_end)) = self.parse_escapes(&(start..i)) {
835                                start..new_end
836                            } else {
837                                return self.invalid("Invalid entity in content", start);
838                            }
839                        } else {
840                            // If escape parsing is not enabled, use the original range
841                            start..i
842                        };
843                        // Add text node with content
844                        self.add_node(parent_idx, NodeType::Text(text_range))?;
845                    }
846                    State::ReadTag
847                }
848                State::End => {
849                    return Ok(());
850                }
851            };
852        }
853
854        Err(ParseXmlError::InvalidXml(
855            "Unexpected end of XML document.".to_string(),
856        ))
857    }
