13/// Information about an XML attribute, storing name and value ranges within the document.
14///
15/// This struct holds references to positions in the source document where the attribute
16/// name and value are located, allowing for efficient string retrieval without copying.
17#[derive(Debug, Clone, PartialEq, Eq)]
18pub struct AttributeInfo {
19    name: XmlRange,
20    value: XmlRange,
21}
22
23impl AttributeInfo {
24    /// Creates a new AttributeInfo with the specified name and value ranges.
25    ///
26    /// # Arguments
27    /// * `name` - The range in the document containing the attribute name
28    /// * `value` - The range in the document containing the attribute value
29    ///
30    /// # Returns
31    /// A new AttributeInfo instance
32    pub fn new(name: XmlRange, value: XmlRange) -> Self {
33        AttributeInfo { name, value }
34    }
35}
36
37/// A reference to an XML attribute within a document.
38///
39/// This struct provides access to an attribute's name and value by combining
40/// a reference to the document with attribute information.
41pub struct Attribute<'a> {
42    doc: &'a Document,
43    data: &'a AttributeInfo,
44}
45
46impl<'a> Attribute<'a> {
47    /// Returns the name of the attribute as a string slice.
48    ///
49    /// # Returns
50    /// A string slice containing the attribute name
51    #[inline]
52    pub fn name(&self) -> &str {
53        self.doc.get_str_from_range(&self.data.name)
54    }
55
56    /// Returns the value of the attribute as a string slice.
57    ///
58    /// # Returns  
59    /// A string slice containing the attribute value
60    #[inline]
61    pub fn value(&self) -> &str {
62        self.doc.get_str_from_range(&self.data.value)
63    }
64}
65
66/// Custom Debug implementation for Attribute that displays name and value.
67impl fmt::Debug for Attribute<'_> {
68    fn fmt(&self, f: &mut fmt::Formatter) -> Result<(), fmt::Error> {
69        write!(
70            f,
71            "Attribute {{ name: \"{}\", value: \"{}\" }}",
72            self.name(),
73            self.value()
74        )
75    }
76}

78/// An iterator over the attributes of an XML node.
79///
80/// This iterator provides access to all attributes belonging to a specific node,
81/// yielding `Attribute` instances for each attribute found.
82#[derive(Clone)]
83pub struct Attributes<'a> {
84    doc: &'a Document,
85    attrs: core::slice::Iter<'a, AttributeInfo>,
86}
87
88impl<'a> Attributes<'a> {
89    /// Creates a new Attributes iterator for the given node.
90    ///
91    /// # Arguments
92    /// * `node` - The node whose attributes should be iterated over
93    ///
94    /// # Returns
95    /// An Attributes iterator that will yield all attributes of the node
96    #[inline]
97    pub fn new(node: &Node<'a>) -> Attributes<'a> {
98        let attrs = match node.node_info.node_type() {
99            NodeType::Element { ref attributes, .. } => {
100                &node.doc.attributes[attributes.start as usize..attributes.end as usize]
101            }
102            _ => &[],
103        };
104
105        Attributes {
106            doc: node.doc,
107            attrs: attrs.iter(),
108        }
109    }
110}
111
112/// Iterator implementation that yields Attribute instances.
113impl<'a> Iterator for Attributes<'a> {
114    type Item = Attribute<'a>;
115
116    /// Returns the next attribute in the iteration.
117    ///
118    /// # Returns
119    /// Some(Attribute) if there are more attributes, None otherwise
120    #[inline]
121    fn next(&mut self) -> Option<Self::Item> {
122        self.attrs.next().map(|attr| Attribute {
123            doc: self.doc,
124            data: attr,
125        })
126    }
127
128    /// Returns the nth attribute, skipping n-1 attributes.
129    ///
130    /// # Arguments
131    /// * `n` - The number of attributes to skip
132    ///
133    /// # Returns
134    /// Some(Attribute) if the nth attribute exists, None otherwise
135    #[inline]
136    fn nth(&mut self, n: usize) -> Option<Self::Item> {
137        self.attrs.nth(n).map(|attr| Attribute {
138            doc: self.doc,
139            data: attr,
140        })
141    }
142
143    /// Returns bounds on the remaining length of the iterator.
144    ///
145    /// # Returns
146    /// A tuple of (lower_bound, upper_bound) where upper_bound is Some(exact_size)
147    #[inline]
148    fn size_hint(&self) -> (usize, Option<usize>) {
149        self.attrs.size_hint()
150    }
151}
152
153/// DoubleEndedIterator implementation allowing reverse iteration.
154impl<'a> DoubleEndedIterator for Attributes<'a> {
155    /// Returns the next attribute from the end of the iteration.
156    ///
157    /// # Returns
158    /// Some(Attribute) if there are more attributes from the back, None otherwise
159    #[inline]
160    fn next_back(&mut self) -> Option<Self::Item> {
161        self.attrs.next_back().map(|attr| Attribute {
162            doc: self.doc,
163            data: attr,
164        })
165    }
166}
