18/// Represents a parsed XML document.
19///
20/// The `Document` struct contains a vector of `NodeInfo` representing the nodes in the document,
21/// a vector of `AttributeInfo` representing the attributes, and the raw XML content as a byte vector.
22/// It provides methods to create a new document from XML content, retrieve the root node,
23/// get nodes by index, add new nodes and attributes, and access the XML content.
24
25#[derive(PartialEq, Eq)]
26pub struct Document {
27    pub nodes: Vec<NodeInfo>,
28    pub attributes: Vec<AttributeInfo>,
29    pub xml: Vec<u8>,
30}
31
32impl Document {
33    /// Creates a new `Document` from the provided XML content.
34    ///
35    /// # Arguments
36    /// - `xml`: A byte vector containing the XML content to be parsed. the Document instance becomes the owner of the XML content
37    ///
38    /// # Returns
39    /// - `Ok(Document)`: If the XML content is successfully parsed and a document is created.
40    /// - `Err(ParseXmlError)`: If there is an error during parsing, such as invalid XML or insufficient memory.
41    ///
42    /// # Example
43    /// ```
44    /// use xhtml_parser::document::Document;
45    /// use xhtml_parser::node::Node;
46    /// let xml_data = b"<root><child>Text</child></root>".to_vec();
47    /// let document = Document::new(xml_data).unwrap();
48    /// let root = document.root().unwrap();
49    /// assert_eq!(root.tag_name(), "root");
50    /// let child = root.first_child().unwrap();
51    /// assert_eq!(child.tag_name(), "child");
52    /// let child_text = child.first_child().unwrap();
53    /// assert_eq!(child_text.text(), "Text");
54    /// ```
55    /// # Notes
56    /// - The `Document` struct is designed to handle XML documents and provides methods for navigating the document tree.
57    /// - The `new` method estimates the number of nodes and attributes based on the XML content and allocates memory accordingly.
58    ///   This is done to optimize performance and reduce memory reallocations during parsing.
59    pub fn new(xml: Vec<u8>) -> Result<Self, ParseXmlError> {
60        // let mut node_count = xml.iter().filter(|&&b| b == b'<').count();
61        // let attr_count = xml.iter().filter(|&&b| b == b'=').count();
62        let mut node_count = memchr_iter(b'<', xml.as_slice()).count();
63        let attr_count = memchr_iter(b'=', xml.as_slice()).count();
64        node_count += (node_count / 10) + 1; // Add 10% buffer for nodes
65
66        if cfg!(feature = "verbose") {
67            println!("Estimated node count: {}", node_count);
68            println!("Estimated attribute count: {}", attr_count);
69        }
70
71        let mut doc = Document {
72            nodes: Vec::with_capacity(node_count + 1), // +1 for root node
73            attributes: Vec::with_capacity(attr_count),
74            xml,
75        };
76        if doc.nodes.capacity() <= node_count || doc.attributes.capacity() < attr_count {
77            return Err(ParseXmlError::NotEnoughMemory);
78        }
79        doc.nodes.push(NodeInfo::new(0, 0, NodeType::Head, 0));
80        doc.parse()?;
81        doc.nodes.shrink_to_fit();
82        doc.attributes.shrink_to_fit();
83
84        if cfg!(feature = "verbose") {
85            println!(
86                "Document created with {} nodes and {} attributes",
87                doc.nodes.len(),
88                doc.attributes.len()
89            );
90        }
91
92        if node_count < doc.nodes.len() {
93            println!(
94                "Warning: Expected {} nodes, but found {}",
95                node_count,
96                doc.nodes.len()
97            );
98        }
99
100        if attr_count < doc.attributes.len() {
101            println!(
102                "Warning: Expected {} attributes, but found {}",
103                attr_count,
104                doc.attributes.len()
105            );
106        }
107
108        Ok(doc)
109    }

111    /// Returns the root node of the document.
112    pub fn root(&self) -> Option<Node> {
113        if self.nodes.len() > 1 {
114            Some(Node::new(1, &self.nodes[1], self))
115        } else {
116            None // No nodes in the document
117        }
118    }

120    /// Retrieves a node by its index.
121    ///
122    /// # Arguments
123    /// - `node_idx`: The index of the node to retrieve.
124    ///
125    /// # Returns
126    /// - `Ok(Node)`: The node at the specified index.
127    /// - `Err(ParseXmlError)`: If the node index is invalid or out of bounds.
128    pub fn get_node(&self, node_idx: NodeIdx) -> Result<Node, ParseXmlError> {
129        if node_idx as usize >= self.nodes.len() {
130            return Err(ParseXmlError::InvalidXml(format!(
131                "Invalid node index: {}",
132                node_idx
133            )));
134        }
135        Ok(Node::new(node_idx, &self.nodes[node_idx as usize], self))
136    }

143    /// Retrieves the parent index of a given node.
144    ///
145    /// # Arguments
146    /// - `node_idx`: The index of the node whose parent index is to be retrieved.
147    ///
148    /// # Returns
149    /// - `Ok(NodeIdx)`: The index of the parent node.
150    /// - `Err(ParseXmlError)`: If the node index is invalid or if the node has no parent (e.g., root node).
151    ///
152    /// # Example
153    /// ```rust
154    /// use xhtml_parser::document::Document;
155    /// use xhtml_parser::node::Node;
156    /// let xml_data = b"<root><child>Text</child></root>".to_vec();
157    /// let document = Document::new(xml_data).unwrap();
158    /// let child_node = document.get_node(2).unwrap(); // Assuming 2 is the index of the child node
159    /// let parent_idx = document.get_parent_idx(child_node.idx()).unwrap();
160    /// assert_eq!(parent_idx, 1); // The parent of the child node is the root node (index 1)
161    /// let root_node = document.root().unwrap();
162    /// assert_eq!(root_node.idx(), 1); // The root node index is 1
163    /// // Attempting to get the parent of the root node should return an error
164    /// let parent_of_root = document.get_parent_idx(root_node.idx());
165    /// assert!(parent_of_root.is_err(), "Root node should not have a parent");
166    /// // Attempting to get the parent of an invalid node index should return an error
167    /// let invalid_node_idx = 100; // Assuming this index is out of bounds
168    /// let invalid_parent = document.get_parent_idx(invalid_node_idx);
169    /// assert!(invalid_parent.is_err(), "Invalid node index should return an error");
170    /// ```
171    /// # Notes
172    /// - The root node (index 1) has no parent, so attempting to get its parent will return an error.
173    /// - The method checks if the node index is valid and returns an error if it is not.
174    /// # Errors
175    /// - `ParseXmlError::InvalidXml`: If the node index is invalid or if the node has no parent (e.g., root node).
176    pub fn get_parent_idx(&self, node_idx: NodeIdx) -> Result<NodeIdx, ParseXmlError> {
177        if (node_idx == 0) || (node_idx as usize >= self.nodes.len()) {
178            return Err(ParseXmlError::InvalidXml(format!(
179                "Invalid node index: {}",
180                node_idx
181            )));
182        }
183
184        if node_idx > 1 {
185            Ok(self.nodes[node_idx as usize].parent_idx().unwrap())
186        } else {
187            return Err(ParseXmlError::InvalidXml(
188                "Root node has no parent".to_string(),
189            ));
190        }
191    }

193    /// Adds a new node to the document.
194    /// This method allows adding a new node to the document tree, setting its parent,
195    /// type, and position in the XML source.
196    ///
197    ///# Arguments
198    /// - `parent_idx`: The index of the parent node.
199    /// - `node_type`: The type of the node to be added (e.g., element, text).
200    /// - `position`: The position of the node in the XML source.
201    ///
202    /// # Returns
203    /// - `Ok(NodeIdx)`: The index of the newly added node.
204    /// - `Err(ParseXmlError)`: If there is an error adding the node (e.g., no more space).
205    pub fn add_node(
206        &mut self,
207        parent_idx: NodeIdx,
208        mut node_type: NodeType,
209        position: XmlIdx,
210    ) -> Result<NodeIdx, ParseXmlError> {
211        let node_idx = self.nodes.len() as NodeIdx;
212
213        if node_idx >= u16::MAX {
214            return Err(ParseXmlError::NoMoreSpace);
215        }
216
217        if let NodeType::Element { attributes, .. } = &mut node_type {
218            *attributes = self.attributes.len() as AttrIdx..self.attributes.len() as AttrIdx;
219        }
220
221        let mut node_info = NodeInfo::new(node_idx, parent_idx, node_type, position);
222        let parent_info = &mut self.nodes[parent_idx as usize];
223
224        if parent_info.first_child_idx() == 0 {
225            parent_info.set_first_child_idx(node_idx); // Set first child if none exists
226            node_info.set_prev_sibling_idx(node_idx); // Set previous sibling to this node (last_child)
227        } else {
228            let first_child_idx = parent_info.first_child_idx(); // Get first child index of parent
229            let last_child_idx = self.nodes[first_child_idx as usize].prev_sibling_idx(); // Get last child Index of parent
230            self.nodes[last_child_idx as usize].set_next_sibling_idx(node_idx); // Set next sibling of last child
231            self.nodes[first_child_idx as usize].set_prev_sibling_idx(node_idx); // Set previous sibling of first child to last child
232            node_info.set_prev_sibling_idx(last_child_idx); // Set previous sibling to last child
233        }
234
235        self.nodes.push(node_info);
236        Ok(node_idx)
237    }

239    /// Adds a new attribute to a node.
240    /// # Arguments
241    /// - `node_idx`: The index of the node to which the attribute will be added.
242    /// - `name`: The name of the attribute as a byte range.
243    /// - `value`: The value of the attribute as a byte range.
244    /// # Returns
245    /// - `Ok(AttrIdx)`: The index of the newly added attribute.
246    /// - `Err(ParseXmlError)`: If there is an error adding the attribute (e.g., node is not an element).
247    pub fn add_attribute(
248        &mut self,
249        node_idx: NodeIdx,
250        name: XmlRange,
251        value: XmlRange,
252    ) -> Result<AttrIdx, ParseXmlError> {
253        let attribute_idx = self.attributes.len() as AttrIdx;
254        self.attributes.push(AttributeInfo::new(name, value));
255        let node_info = &mut self.nodes[node_idx as usize];
256        if !node_info.is_element() {
257            return Err(ParseXmlError::InternalError);
258        } else {
259            let mut attributes_range = match &node_info.node_type() {
260                NodeType::Element { attributes, .. } => attributes.clone(),
261                _ => return Err(ParseXmlError::InternalError),
262            };
263            attributes_range.end += 1; // Extend the range to include the new attribute
264            node_info.set_node_type(NodeType::Element {
265                name: match &node_info.node_type() {
266                    NodeType::Element { name, .. } => name.clone(),
267                    _ => return Err(ParseXmlError::InternalError),
268                },
269                attributes: attributes_range,
270            });
271        }
272        Ok(attribute_idx)
273    }

275    /// Retrieves a string slice from the XML content based on the given range.
276    /// # Arguments
277    /// - `range`: A reference to an `XmlRange` that specifies the start and end indices of the desired substring.
278    /// # Returns
279    /// - `&str`: A string slice containing the XML content from the specified range.
280    pub fn get_str_from_range(&self, range: &XmlRange) -> &str {
281        let xml_content = &self.xml[range.start as usize..range.end as usize];
282        std::str::from_utf8(xml_content).unwrap_or("non valid utf-8")
283    }

286impl<'input> fmt::Debug for Document {
287    fn fmt(&self, f: &mut fmt::Formatter) -> Result<(), fmt::Error> {
288        if let Some(root) = self.root() {
289            // write!(f, "Document [{}]", root.tag_name())?;
290
291            macro_rules! writeln_indented {
292                ($indent:expr, $f:expr, $fmt:expr) => {
293                    for _ in 0..$indent { write!($f, "    ")?; }
294                    writeln!($f, $fmt)?;
295                };
296
297                ($indent:expr, $f:expr, $fmt:expr, $($arg:tt)*) => {
298                    for _ in 0..$indent { write!($f, "    ")?; }
299                    writeln!($f, $fmt, $($arg)*)?;
300                };
301            }
302
303            fn print_into_iter<
304                T: fmt::Debug,
305                E: ExactSizeIterator<Item = T>,
306                I: IntoIterator<Item = T, IntoIter = E>,
307            >(
308                prefix: &str,
309                data: I,
310                indent: usize,
311                f: &mut fmt::Formatter,
312            ) -> Result<(), fmt::Error> {
313                let data = data.into_iter();
314
315                if data.len() == 0 {
316                    return Ok(());
317                }
318
319                writeln_indented!(indent, f, "{}: [", prefix);
320                for v in data {
321                    writeln_indented!(indent + 1, f, "{:?}", v);
322                }
323                writeln_indented!(indent, f, "]");
324                Ok(())
325            }
326
327            fn print_node(
328                node: Node,
329                indent: usize,
330                f: &mut fmt::Formatter,
331            ) -> Result<(), fmt::Error> {
332                if node.is_element() {
333                    writeln_indented!(indent, f, "Element {{");
334                    writeln_indented!(indent, f, "    tag_name: {:?}", node.tag_name());
335                    print_into_iter("attributes", node.attributes(), indent + 1, f)?;
336
337                    if node.has_children() {
338                        writeln_indented!(indent, f, "    children: [");
339                        print_children(node, indent + 2, f)?;
340                        writeln_indented!(indent, f, "    ]");
341                    }
342
343                    writeln_indented!(indent, f, "}}");
344                } else if node.is_text() {
345                    writeln_indented!(indent, f, "Text {{");
346                    writeln_indented!(indent, f, "    \"{}\"", node.text());
347                    writeln_indented!(indent, f, "}}");
348                    //writeln_indented!(indent, f, "{:?}", node);
349                    // } else if node.is_root() {
350                    //     writeln_indented!(indent, f, "Root {{}}");
351                } else {
352                    writeln_indented!(indent, f, "Unknown Node!");
353                }
354                Ok(())
355            }
356
357            fn print_children(
358                parent: Node,
359                indent: usize,
360                f: &mut fmt::Formatter,
361            ) -> Result<(), fmt::Error> {
362                for child in parent.children() {
363                    print_node(child, indent, f)?;
364                }
365
366                Ok(())
367            }
368
369            writeln!(f, "Document [")?;
370            print_node(root, 1, f)?;
371            writeln!(f, "]")?;
372
373            Ok(())
374        } else {
375            write!(f, "Document [No root node]")?;
376            Ok(())
377        }
378    }
